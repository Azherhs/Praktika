#include <iostream>
using namespace std;
double *Pamat(int kol) {
    double *num= new double[kol];
    return num;
}
double* Sorti(double* num, int kol){
    for(int i=0;i<kol;i++){
        if (i%2==0) swap(num[i],num[i+1]);
    }
    return num;
}
double* Zapis(double* num,int kol) {
    for (int i = 0; i < kol; i++) {
        cin >> num[i];
    }
    return num;
}
void vivod(double* num, int kol){
    for(int i=0; i<kol; i++) {
        cout << num[i]<< " ";
    }
}
double* Ochistka(double* num) {
    delete[]num;
    return num;
}
